## 题目分析
程序里有4个变量 ， 为a , b , c , d .
3个老师猜测:
  甲：a同学第1 , b同学第3 .
   
  乙：c同学第1 , d同学第3 .
  
  丙：d同学第1 , b同学第3 .
  
根据3位老师说的话提出我们要用4个循环控制名词

分别以：a , b , c , d ;

三个同学的名词来控制循环次数：

  for ( a = 1 ; a <= 4 ; a ++ );
  
  for ( b = 1 ; b <= 4 ; b ++ );
  
  for ( c = 1 ; c <= 4 ; c ++ );
  
  for ( d = 1 ; d <= 4 ; d ++ );
  
## 代码如下：
```
#include <stdio.h>
#include <stdlib.h>

int main()
{
	int a , b , c , d ;
	for ( a = 1 ; a <= 4 ; a ++ )
	{
		for ( b = 1 ; b <= 4 ; b ++ )
		{
			for ( c = 1 ; c <= 4 ; c ++ )
			{
				for ( d = 1 ; d <= 4 ; d ++ )
				{
					if ( a + b + c + d != 10 || a * b * c * d != 24 )
					{
						continue ;
					}					
					if (  ( ( a == 1) + (b == 3) == 1 ) &&
						  ( ( c == 1) + (d == 3) == 1 ) &&
						  ( ( d == 1) + (b == 3) == 1 ) )
					{
						printf("a = %d , b = %d , c = %d , d = %d .\n", a , b , c , d );
					}
				}
			}
		}
	}
	return 0 ;
}
```
